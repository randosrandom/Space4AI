278  SystemData::initialize_resources(const nl::json& resources_json)
279  {
280  std::size_t res_idx = 0;
281 
282  for(const auto& [cl, data]: resources_json.items())
283  {
284  this->cls[ResIdxFromType(Type)].emplace_back(cl, Type);
285 
286  for(const auto& [res_name, res_data]: data.items())
287  {
288  std::string description = "No description available";
289 
290  std::size_t n_cores = 1;
291 
292  if(res_data.contains("description"))
293  description = res_data.at("description").template get<std::string>();
294  if(res_data.contains("n_cores"))
295  n_cores = res_data.at("n_cores").template get<size_t>();
296 
297  Resource<Type> res(
298  static_cast<std::string>(res_name),
299  description,
300  static_cast<std::string>(cl),
301  res_data.at("cost").template get<double>(),
302  res_data.at("memory").template get<double>(),
303  res_data.at("number").template get<size_t>(),
304  n_cores
305  );
306 
307  this->res_name_to_type_and_idx.emplace(
308  res_name,
309  std::make_pair(Type, res_idx)
310  );
311 
312  this->cls[ResIdxFromType(Type)].back().add_resource(res_idx++);
313 
314  this->all_resources.add_resource(std::move(res));
315  }
316  }
