350  void
351  AllResources::add_resource(T&& resource)
352  {
353  if constexpr(
354  std::is_same_v<T, Resource<ResourceType::Edge>> // passed by lvalue
355  ||
356  std::is_same_v<T, Resource<ResourceType::Edge>&&>) // passed by rvalue
357  {
358  edge_resources.emplace_back(std::forward<T>(resource));
359  number_resources[ResIdxFromType(ResourceType::Edge)]++;
360  }
361 
362  else if constexpr(
363  std::is_same_v<T, Resource<ResourceType::VM>>

366  {
367  vm_resources.emplace_back(std::forward<T>(resource));
368  number_resources[ResIdxFromType(ResourceType::VM)]++;

370

373  ||

375  {
376  faas_resources.emplace_back(std::forward<T>(resource));
